import network import urequests import ujson import time from machine import Pin, PWM # 🔐 WiFi 정보 입력 ssid = "와이파이이름" password = "와이파이비번" # 🌐 Firebase URL url = "https://pico-1a284-default-rtdb.asia-southeast1.firebasedatabase.app/ledStatus.json" # 📡 WiFi 연결 wifi = network.WLAN(network.STA_IF) wifi.active(True) wifi.connect(ssid, password) print("WiFi 연결 중...") while not wifi.isconnected(): time.sleep(1) print("WiFi 연결 완료:", wifi.ifconfig()) # 🎨 RGB 핀 설정 (네가 연결한 그대로) r = PWM(Pin(0)) # GP0 g = PWM(Pin(1)) # GP1 b = PWM(Pin(2)) # GP2 # PWM 주파수 설정 r.freq(1000) g.freq(1000) b.freq(1000) # 🌈 색 설정 함수 def set_color(rv, gv, bv): r.duty_u16(int(rv * 257)) # 0~255 → 0~65535 변환 g.duty_u16(int(gv * 257)) b.duty_u16(int(bv * 257)) # 🔁 반복 실행 while True: try: response = urequests.get(url) data = ujson.loads(response.text) response.close() # 데이터가 정상일 때 if isinstance(data, dict): rv = data.get("r", 0) gv = data.get("g", 0) bv = data.get("b", 0) print("RGB:", rv, gv, bv) set_color(rv, gv, bv) except Exception as e: print("에러:", e) time.sleep(1)
